=====================

You've caught two of your fellow minions passing coded notes back and forth -- while they're on duty, no less! public class Solution {
    public static String solution(String x) {
        // Your code here
        StringBuilder SolString=new StringBuilder();
        int CharAscii='\0';
        int i=0;
        while(i<x.length()){
            CharAscii=x.charAt(i);
            if(CharAscii >= 97 && CharAscii <= 122){
                SolString.append((char) (122 - CharAscii  + 97));
            }
            else{
                SolString.append(x.charAt(i));
            }
             i++;
        }
        SolString.append('\0');
        return SolString.toString();
    }
}
